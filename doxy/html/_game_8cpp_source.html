65  Piece* nextPiece = pieceFactory->getPiece();

68 
69  //setting the scoreboard

72  resourceManager,
73  1,

75  X_OFFSET + (this->boardWidth + 2) * 32,

83  resourceManager,

89  //setting the fall sound

100 
101  //setting the main music
102  sf::Music* music = resourceManager->getMusic("tetris.ogg");
103  music->setVolume(this->volume);
104  music->setLoop(true);
105  music->play();
106 
107  //setting game flags
108  bool fastFallFlag = false;
109  bool pausedFlag = false;
110 
111  //setting clocks used for game's time counting
112  sf::Clock frameClock;
113  sf::Clock keyClock;
114  sf::Time frameTime = frameClock.getElapsedTime();
115  sf::Time keyTime = keyClock.getElapsedTime();
116 
117  //main game loop
118  while (window->isOpen()) {
119 
120  //entering main menu
121  if (*gameStatePtr == MENU) {
122  music->stop();
123  break;
124  }
125 
126  //game over
127  if (gameOver) {
128  *gameStatePtr = GAMEOVER;
129  *levelPtr = level;
130  *scorePtr = score;
131 
132  music->stop();
133 
134  delete currentPiece;
135  delete nextPiece;
136  delete ghostPiece;
137 
138  break;
139  }
140 
141  //event handling
142  sf::Event event;
143  while (window->pollEvent(event)) {
144  if (event.type == sf::Event::Closed)
145  window->close();
146  else if (event.type == sf::Event::KeyPressed) {
147  if (event.key.code == sf::Keyboard::Up) {
148  if (!pausedFlag)
149  rotateLeft(currentPiece, ghostPiece);
150  }
151  else if (event.key.code == sf::Keyboard::Down) {
152  if (!pausedFlag)
153  rotateRight(currentPiece, ghostPiece);
154  }
155  else if (event.key.code == sf::Keyboard::Escape)
156  *gameStatePtr = MENU;
157  else if (event.key.code == sf::Keyboard::P)
158  pausedFlag = !pausedFlag;
159  }
160  }
161 
162  //event handling for continuuous key entering
163  //allows to achieve more smooth game input
164  keyTime = keyClock.getElapsedTime();
165  if (!pausedFlag) {
166  if (sf::Keyboard::isKeyPressed(sf::Keyboard::Space) && keyTime.asSeconds() > 0.04) {
167  fastFallFlag = true;
168  keyClock.restart();
169  } else if (sf::Keyboard::isKeyPressed(sf::Keyboard::Left) && keyTime.asSeconds() > 0.09) { //0.08
170  moveLeft(currentPiece, ghostPiece);
171  keyClock.restart();
172  } else if (sf::Keyboard::isKeyPressed(sf::Keyboard::Right) && keyTime.asSeconds() > 0.09) {
173  moveRight(currentPiece, ghostPiece);
174  keyClock.restart();
175  }
176  }
177 
178  //main SFML window drawing sequence
179  window->clear();
180  window->draw(background);
181  window->draw(nextBoard);
182  window->draw(scoreBoard);
183  window->draw(*gameBoard);
184  window->draw(*currentPiece);
185  if (ghostFlag)
186  window->draw(*ghostPiece);
187  if (pausedFlag)
188  window->draw(pausedText);
189  window->display();
190 
191  //main game condition system, falling of the pieces
192  frameTime = frameClock.getElapsedTime();
193  if (!pausedFlag && (fastFallFlag | (frameTime.asSeconds() > 0.5 - (level - 1) * 0.025))) {
194  frameClock.restart();
195 
196  //if fast falling, increment the score
197  if (fastFallFlag)
198  this->score += 1;
199 
200  if (!fallDown(currentPiece)) {
201  fallSound.play();
202  gameOver = !gameBoard->add(currentPiece);
203 
204  delete currentPiece;
205  delete ghostPiece;
206 
207  //swap pieces
208  currentPiece = nextPiece;
209  currentPiece->setPiecePosition(this->boardWidth / 2 - 1, DEFAULT_Y);
210  //get new piece as the next piece
211  nextPiece = pieceFactory->getPiece();
212  nextBoard.setPiece(nextPiece);
213  //get new ghost piece and update it's position
214  ghostPiece = pieceFactory->getGhostPiece(currentPiece);
215 
216  //getting number of cleared rows
217  int clearedRows = gameBoard->updateBoard();
218 
219  if (clearedRows)
220  clearedSound.play();
221 
222  //setting ghost position after board updating
223  setGhostPosition(currentPiece, ghostPiece);
224 
225  //update score fields
226  this->totalRows += clearedRows;
227  updateScore(clearedRows);
228  updateLevel();
229 
230  }
231  fastFallFlag = false;
232 
233  //update scoreboard
234  scoreBoard.setScore(this->score);
235  scoreBoard.setLevel(this->level);
236  }
237  }

318 void Game::setGhostPosition(Piece *currentPiece, Piece* ghostPiece) {

321  //set current ghost position the same as the current piece position
