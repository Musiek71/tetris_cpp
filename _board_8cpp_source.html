61 bool Board::add(Piece *piece) {
62  //for every point of the piece's shape, add the point to the board
63  //if point we are adding is in the spawning zone (Y less than 3), return false
64  //returning false means the game is over
65  for (int i = 0; i < 4; i++) {
66  if (piece->getPiecePosition().getY() + piece->getCurrentShape()[i].getY() < 3)
67  return false;
68  //update the board with piece's shape

70  //update the board's textures

85 int Board::updateBoard() {
86  bool foundFullRow = true;
87  int lineCounter = 0;
88 
89  //beginning from the bottom, get all full rows, and remove them by pushing the board down by one row.
90  for (int y = this->boardHeight - 2; y > 0; y--) {
91  foundFullRow = true;
92  for (int x = 1; x < this->boardWidth - 1; x++) {
93  //if any of the row's column is empty, the row isn't full
94  if (board[x][y] == 0)
95  foundFullRow = false;
96  }
97  if (foundFullRow) {
98  pushRowDown(y);
99  lineCounter++;
100  y++;
101  }
102  }
103  return lineCounter;
104 }

116 void Board::updateAllTextures(int tileSize) {

118  //editing j here modifies the number of visible walls, default is 3 rows for piece spawning.
119  //DEFAULT_Y_OFFSET is used here to skip drawing few top rows used only for piece spawning.
120  for (int j = DEFAULT_Y_OFFSET / 32; j < this->boardHeight; j++) {
121  //get the tileNumber
122  int tileNumber = board[i][j];
123 
124  //get the pointer to the current quad

126 
127  //define its 4 corners
128  quad[0].position = sf::Vector2f(i * tileSize, j * tileSize);
129  quad[1].position = sf::Vector2f((i + 1) * tileSize, j * tileSize);
130  quad[2].position = sf::Vector2f((i + 1) * tileSize, (j + 1) * tileSize);
131  quad[3].position = sf::Vector2f(i * tileSize, (j + 1) * tileSize);
132 
133  //define its 4 texture coordinates
134  quad[0].texCoords = sf::Vector2f(tileNumber * tileSize, 0);
135  quad[1].texCoords = sf::Vector2f((tileNumber + 1) * tileSize, 0);
136  quad[2].texCoords = sf::Vector2f((tileNumber + 1) * tileSize, tileSize);
