83  while (window->isOpen()) {

89 
90  sf::Event event;
91  while (window->pollEvent(event)) {
92  if (event.type == sf::Event::Closed)
93  window->close();
94  if (event.type == sf::Event::TextEntered) {
95  if (event.text.unicode == '\b') {
96  if (nick.length() > 0)
97  nick.erase(nick.length() - 1, 1);
98  }
99  else if (event.text.unicode != ' ' && event.text.unicode >= 'A' && event.text.unicode <= 'z') {
100  if (nick.length() < 9)
101  nick += event.text.unicode;
102  }
103  nickText.setString("Nick:" + nick);
104  nickText.setPosition(window->getSize().x / 2 - nickText.getGlobalBounds().width / 2, window->getSize().y / 8 * 3);
105  }
106  }
107 
108  sf::Vector2f mouseViewPos = window->mapPixelToCoords(sf::Mouse::getPosition(*window));
109 
110  if (menuButton->updateButton(mouseViewPos)) {
111  *gameStatePtr = MENU;
112  }
113 
114  if (gameButton->updateButton(mouseViewPos)) {
115  *gameStatePtr = GAME;
116  }
117 
118  if (leaderboardButton->updateButton(mouseViewPos)) {
119  *gameStatePtr = LEADERBOARD;
120  }
121 
122  window->clear();
123  window->draw(background);
124 
125  window->draw(topText);
126  window->draw(scoreText);
127  window->draw(levelText);
128  window->draw(nickText);
129 
130  window->draw(*menuButton);
